186 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters, BaseType_t xRunPrivileged )
187 {
188  /* Simulate the stack frame as it would be created by a context switch
189  interrupt. */
190  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
191  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
192  pxTopOfStack--;
193  *pxTopOfStack = ( ( StackType_t ) pxCode ) & portSTART_ADDRESS_MASK; /* PC */
194  pxTopOfStack--;
195  *pxTopOfStack = 0; /* LR */
196  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
197  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
198 
199  /* A save method is being used that requires each task to maintain its
200  own exec return value. */
201  pxTopOfStack--;
202  *pxTopOfStack = portINITIAL_EXC_RETURN;
203 
204  pxTopOfStack -= 9; /* R11, R10, R9, R8, R7, R6, R5 and R4. */
205 
206  if( xRunPrivileged == pdTRUE )
207  {
208  *pxTopOfStack = portINITIAL_CONTROL_IF_PRIVILEGED;
209  }
210  else
211  {
212  *pxTopOfStack = portINITIAL_CONTROL_IF_UNPRIVILEGED;
213  }
214 
215  return pxTopOfStack;
216 }

238 static void prvSVCHandler( uint32_t *pulParam )
239 {
240 uint8_t ucSVCNumber;
241 
242  /* The stack contains: r0, r1, r2, r3, r12, r14, the return address and
243  xPSR. The first argument (r0) is pulParam[ 0 ]. */
244  ucSVCNumber = ( ( uint8_t * ) pulParam[ portOFFSET_TO_PC ] )[ -2 ];
245  switch( ucSVCNumber )
246  {
247  case portSVC_START_SCHEDULER : portNVIC_SYSPRI1_REG |= portNVIC_SVC_PRI;
248  prvRestoreContextOfFirstTask();
249  break;
250 
251  case portSVC_YIELD : portNVIC_INT_CTRL_REG = portNVIC_PENDSVSET_BIT;
252  /* Barriers are normally not required
253  but do ensure the code is completely
254  within the specified behaviour for the
255  architecture. */
256  __asm volatile( "dsb" ::: "memory" );
257  __asm volatile( "isb" );
258 
259  break;
260 
261  case portSVC_RAISE_PRIVILEGE : __asm volatile
262  (
263  " mrs r1, control \n" /* Obtain current control value. */
264  " bic r1, #1 \n" /* Set privilege bit. */
265  " msr control, r1 \n" /* Write back new control value. */
266  ::: "r1", "memory"
267  );
268  break;
269 
270  default : /* Unknown SVC call. */
271  break;
272  }
273 }

276 static void prvRestoreContextOfFirstTask( void )
277 {
278  __asm volatile
279  (
280  " ldr r0, =0xE000ED08 \n" /* Use the NVIC offset register to locate the stack. */
281  " ldr r0, [r0] \n"
282  " ldr r0, [r0] \n"
283  " msr msp, r0 \n" /* Set the msp back to the start of the stack. */
284  " ldr r3, pxCurrentTCBConst2 \n" /* Restore the context. */
285  " ldr r1, [r3] \n"
286  " ldr r0, [r1] \n" /* The first item in the TCB is the task top of stack. */
287  " add r1, r1, #4 \n" /* Move onto the second item in the TCB... */
288  " ldr r2, =0xe000ed9c \n" /* Region Base Address register. */
289  " ldmia r1!, {r4-r11} \n" /* Read 4 sets of MPU registers. */
290  " stmia r2!, {r4-r11} \n" /* Write 4 sets of MPU registers. */
291  " ldmia r0!, {r3-r11, r14} \n" /* Pop the registers that are not automatically saved on exception entry. */
292  " msr control, r3 \n"
293  " msr psp, r0 \n" /* Restore the task stack pointer. */
294  " mov r0, #0 \n"
295  " msr basepri, r0 \n"
296  " bx r14 \n"
297  " \n"
298  " .align 4 \n"
299  "pxCurrentTCBConst2: .word pxCurrentTCB \n"
300  );
301 }

307 BaseType_t xPortStartScheduler( void )
308 {
309  /* configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to 0. See
310  http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html */
311  configASSERT( ( configMAX_SYSCALL_INTERRUPT_PRIORITY ) );
312 
313  #if( configASSERT_DEFINED == 1 )
314  {
315  volatile uint32_t ulOriginalPriority;
316  volatile uint8_t * const pucFirstUserPriorityRegister = ( volatile uint8_t * const ) ( portNVIC_IP_REGISTERS_OFFSET_16 + portFIRST_USER_INTERRUPT_NUMBER );
317  volatile uint8_t ucMaxPriorityValue;
318 
319  /* Determine the maximum priority from which ISR safe FreeRTOS API
320  functions can be called. ISR safe functions are those that end in
321  "FromISR". FreeRTOS maintains separate thread and ISR API functions to
322  ensure interrupt entry is as fast and simple as possible.
323 
324  Save the interrupt priority value that is about to be clobbered. */
325  ulOriginalPriority = *pucFirstUserPriorityRegister;
326 
327  /* Determine the number of priority bits available. First write to all
328  possible bits. */
329  *pucFirstUserPriorityRegister = portMAX_8_BIT_VALUE;
330 
331  /* Read the value back to see how many bits stuck. */
332  ucMaxPriorityValue = *pucFirstUserPriorityRegister;
333 
334  /* Use the same mask on the maximum system call priority. */
335  ucMaxSysCallPriority = configMAX_SYSCALL_INTERRUPT_PRIORITY & ucMaxPriorityValue;
336 
337  /* Calculate the maximum acceptable priority group value for the number
338  of bits read back. */
339  ulMaxPRIGROUPValue = portMAX_PRIGROUP_BITS;
340  while( ( ucMaxPriorityValue & portTOP_BIT_OF_BYTE ) == portTOP_BIT_OF_BYTE )
341  {
342  ulMaxPRIGROUPValue--;
343  ucMaxPriorityValue <<= ( uint8_t ) 0x01;
344  }
345 
346  #ifdef __NVIC_PRIO_BITS
347  {
348  /* Check the CMSIS configuration that defines the number of
349  priority bits matches the number of priority bits actually queried
350  from the hardware. */
351  configASSERT( ( portMAX_PRIGROUP_BITS - ulMaxPRIGROUPValue ) == __NVIC_PRIO_BITS );
352  }
353  #endif
354 
355  #ifdef configPRIO_BITS
356  {
357  /* Check the FreeRTOS configuration that defines the number of
358  priority bits matches the number of priority bits actually queried
359  from the hardware. */
360  configASSERT( ( portMAX_PRIGROUP_BITS - ulMaxPRIGROUPValue ) == configPRIO_BITS );
361  }
362  #endif
363 
364  /* Shift the priority group value back to its position within the AIRCR
365  register. */
366  ulMaxPRIGROUPValue <<= portPRIGROUP_SHIFT;
367  ulMaxPRIGROUPValue &= portPRIORITY_GROUP_MASK;
368 
369  /* Restore the clobbered interrupt priority register to its original
370  value. */
371  *pucFirstUserPriorityRegister = ulOriginalPriority;
372  }
373  #endif /* conifgASSERT_DEFINED */
374 
375  /* Make PendSV and SysTick the same priority as the kernel, and the SVC
376  handler higher priority so it can be used to exit a critical section (where
377  lower priorities are masked). */
378  portNVIC_SYSPRI2_REG |= portNVIC_PENDSV_PRI;
379  portNVIC_SYSPRI2_REG |= portNVIC_SYSTICK_PRI;
380 
381  /* Configure the regions in the MPU that are common to all tasks. */
382  prvSetupMPU();
383 
384  /* Start the timer that generates the tick ISR. Interrupts are disabled
385  here already. */
386  vPortSetupTimerInterrupt();
387 
388  /* Initialise the critical nesting count ready for the first task. */
389  uxCriticalNesting = 0;
390 
391  /* Ensure the VFP is enabled - it should be anyway. */
392  vPortEnableVFP();
393 
394  /* Lazy save always. */
395  *( portFPCCR ) |= portASPEN_AND_LSPEN_BITS;
396 
397  /* Start the first task. This also clears the bit that indicates the FPU is
398  in use in case the FPU was used before the scheduler was started - which
399  would otherwise result in the unnecessary leaving of space in the SVC stack
400  for lazy saving of FPU registers. */
401  __asm volatile(
402  " ldr r0, =0xE000ED08 \n" /* Use the NVIC offset register to locate the stack. */
403  " ldr r0, [r0] \n"
404  " ldr r0, [r0] \n"
405  " msr msp, r0 \n" /* Set the msp back to the start of the stack. */
406  " mov r0, #0 \n" /* Clear the bit that indicates the FPU is in use, see comment above. */
407  " msr control, r0 \n"
408  " cpsie i \n" /* Globally enable interrupts. */
409  " cpsie f \n"
410  " dsb \n"
411  " isb \n"
412  " svc %0 \n" /* System call to start first task. */
413  " nop \n"
414  :: "i" (portSVC_START_SCHEDULER) : "memory" );
415 
416  /* Should not get here! */
417  return 0;
418 }

454 void xPortPendSVHandler( void )
455 {
456  /* This is a naked function. */
457 
458  __asm volatile
459  (
460  " mrs r0, psp \n"
461  " isb \n"
462  " \n"
463  " ldr r3, pxCurrentTCBConst \n" /* Get the location of the current TCB. */
464  " ldr r2, [r3] \n"
465  " \n"
466  " tst r14, #0x10 \n" /* Is the task using the FPU context? If so, push high vfp registers. */
467  " it eq \n"
468  " vstmdbeq r0!, {s16-s31} \n"
469  " \n"
470  " mrs r1, control \n"
471  " stmdb r0!, {r1, r4-r11, r14} \n" /* Save the remaining registers. */
472  " str r0, [r2] \n" /* Save the new top of stack into the first member of the TCB. */
473  " \n"
474  " stmdb sp!, {r0, r3} \n"
475  " mov r0, %0 \n"
476  " msr basepri, r0 \n"
477  " dsb \n"
478  " isb \n"
479  " bl vTaskSwitchContext \n"
480  " mov r0, #0 \n"
481  " msr basepri, r0 \n"
482  " ldmia sp!, {r0, r3} \n"
483  " \n" /* Restore the context. */
484  " ldr r1, [r3] \n"
485  " ldr r0, [r1] \n" /* The first item in the TCB is the task top of stack. */
486  " add r1, r1, #4 \n" /* Move onto the second item in the TCB... */
487  " ldr r2, =0xe000ed9c \n" /* Region Base Address register. */
488  " ldmia r1!, {r4-r11} \n" /* Read 4 sets of MPU registers. */
489  " stmia r2!, {r4-r11} \n" /* Write 4 sets of MPU registers. */
490  " ldmia r0!, {r3-r11, r14} \n" /* Pop the registers that are not automatically saved on exception entry. */
491  " msr control, r3 \n"
492  " \n"
493  " tst r14, #0x10 \n" /* Is the task using the FPU context? If so, pop the high vfp registers too. */
494  " it eq \n"
495  " vldmiaeq r0!, {s16-s31} \n"
496  " \n"
497  " msr psp, r0 \n"
498  " bx r14 \n"
499  " \n"
500  " .align 4 \n"
501  "pxCurrentTCBConst: .word pxCurrentTCB \n"
502  ::"i"(configMAX_SYSCALL_INTERRUPT_PRIORITY)
503  );
504 }

618 static uint32_t prvGetMPURegionSizeSetting( uint32_t ulActualSizeInBytes )
619 {
620 uint32_t ulRegionSize, ulReturnValue = 4;
621 
622  /* 32 is the smallest region size, 31 is the largest valid value for
623  ulReturnValue. */
624  for( ulRegionSize = 32UL; ulReturnValue < 31UL; ( ulRegionSize <<= 1UL ) )
625  {
626  if( ulActualSizeInBytes <= ulRegionSize )
627  {
628  break;
629  }
630  else
631  {
632  ulReturnValue++;
633  }
634  }
635 
636  /* Shift the code by one before returning so it can be written directly
637  into the the correct bit position of the attribute register. */
638  return ( ulReturnValue << 1UL );
639 }

660 void vPortStoreTaskMPUSettings( xMPU_SETTINGS *xMPUSettings, const struct xMEMORY_REGION * const xRegions, StackType_t *pxBottomOfStack, uint32_t ulStackDepth )
661 {
662 extern uint32_t __SRAM_segment_start__[];
663 extern uint32_t __SRAM_segment_end__[];
664 extern uint32_t __privileged_data_start__[];
665 extern uint32_t __privileged_data_end__[];

667 uint32_t ul;

669  if( xRegions == NULL )
670  {
671  /* No MPU regions are specified so allow access to all RAM. */
672  xMPUSettings->xRegion[ 0 ].ulRegionBaseAddress =
673  ( ( uint32_t ) __SRAM_segment_start__ ) | /* Base address. */
674  ( portMPU_REGION_VALID ) |
675  ( portSTACK_REGION );
676 
677  xMPUSettings->xRegion[ 0 ].ulRegionAttribute =
678  ( portMPU_REGION_READ_WRITE ) |
679  ( portMPU_REGION_CACHEABLE_BUFFERABLE ) |
680  ( prvGetMPURegionSizeSetting( ( uint32_t ) __SRAM_segment_end__ - ( uint32_t ) __SRAM_segment_start__ ) ) |
681  ( portMPU_REGION_ENABLE );
682 
683  /* Re-instate the privileged only RAM region as xRegion[ 0 ] will have
684  just removed the privileged only parameters. */
685  xMPUSettings->xRegion[ 1 ].ulRegionBaseAddress =
686  ( ( uint32_t ) __privileged_data_start__ ) | /* Base address. */
687  ( portMPU_REGION_VALID ) |
688  ( portSTACK_REGION + 1 );
689 
690  xMPUSettings->xRegion[ 1 ].ulRegionAttribute =
691  ( portMPU_REGION_PRIVILEGED_READ_WRITE ) |
692  ( portMPU_REGION_CACHEABLE_BUFFERABLE ) |
693  prvGetMPURegionSizeSetting( ( uint32_t ) __privileged_data_end__ - ( uint32_t ) __privileged_data_start__ ) |
694  ( portMPU_REGION_ENABLE );
695 
696  /* Invalidate all other regions. */
697  for( ul = 2; ul <= portNUM_CONFIGURABLE_REGIONS; ul++ )
698  {
699  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress = ( portSTACK_REGION + ul ) | portMPU_REGION_VALID;
700  xMPUSettings->xRegion[ ul ].ulRegionAttribute = 0UL;
701  }
702  }
703  else
704  {
705  /* This function is called automatically when the task is created - in
706  which case the stack region parameters will be valid. At all other
707  times the stack parameters will not be valid and it is assumed that the
708  stack region has already been configured. */
709  if( ulStackDepth > 0 )
710  {
711  /* Define the region that allows access to the stack. */
712  xMPUSettings->xRegion[ 0 ].ulRegionBaseAddress =
713  ( ( uint32_t ) pxBottomOfStack ) |
714  ( portMPU_REGION_VALID ) |
715  ( portSTACK_REGION ); /* Region number. */
716 
717  xMPUSettings->xRegion[ 0 ].ulRegionAttribute =
718  ( portMPU_REGION_READ_WRITE ) | /* Read and write. */
719  ( prvGetMPURegionSizeSetting( ulStackDepth * ( uint32_t ) sizeof( StackType_t ) ) ) |
720  ( portMPU_REGION_CACHEABLE_BUFFERABLE ) |
721  ( portMPU_REGION_ENABLE );
722  }
723 
724  lIndex = 0;
725 
726  for( ul = 1; ul <= portNUM_CONFIGURABLE_REGIONS; ul++ )
727  {
728  if( ( xRegions[ lIndex ] ).ulLengthInBytes > 0UL )
729  {
730  /* Translate the generic region definition contained in
731  xRegions into the CM3 specific MPU settings that are then
732  stored in xMPUSettings. */
733  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress =
734  ( ( uint32_t ) xRegions[ lIndex ].pvBaseAddress ) |
735  ( portMPU_REGION_VALID ) |
736  ( portSTACK_REGION + ul ); /* Region number. */
737 
738  xMPUSettings->xRegion[ ul ].ulRegionAttribute =
739  ( prvGetMPURegionSizeSetting( xRegions[ lIndex ].ulLengthInBytes ) ) |
740  ( xRegions[ lIndex ].ulParameters ) |
741  ( portMPU_REGION_ENABLE );
742  }
743  else
744  {
745  /* Invalidate the region. */
746  xMPUSettings->xRegion[ ul ].ulRegionBaseAddress = ( portSTACK_REGION + ul ) | portMPU_REGION_VALID;
747  xMPUSettings->xRegion[ ul ].ulRegionAttribute = 0UL;
748  }
749 
750  lIndex++;
751  }
752  }

758  void vPortValidateInterruptPriority( void )
759  {
760  uint32_t ulCurrentInterrupt;
761  uint8_t ucCurrentPriority;
762 
763  /* Obtain the number of the currently executing interrupt. */
764  __asm volatile( "mrs %0, ipsr" : "=r"( ulCurrentInterrupt ) :: "memory" );
765 
766  /* Is the interrupt number a user defined interrupt? */
767  if( ulCurrentInterrupt >= portFIRST_USER_INTERRUPT_NUMBER )
768  {
769  /* Look up the interrupt's priority. */
770  ucCurrentPriority = pcInterruptPriorityRegisters[ ulCurrentInterrupt ];
771 
772  /* The following assertion will fail if a service routine (ISR) for
773  an interrupt that has been assigned a priority above
774  configMAX_SYSCALL_INTERRUPT_PRIORITY calls an ISR safe FreeRTOS API
775  function. ISR safe FreeRTOS API functions must *only* be called
776  from interrupts that have been assigned a priority at or below
777  configMAX_SYSCALL_INTERRUPT_PRIORITY.
778 
779  Numerically low interrupt priority numbers represent logically high
780  interrupt priorities, therefore the priority of the interrupt must
781  be set to a value equal to or numerically *higher* than
782  configMAX_SYSCALL_INTERRUPT_PRIORITY.
783 
784  Interrupts that use the FreeRTOS API must not be left at their
785  default priority of zero as that is the highest possible priority,
786  which is guaranteed to be above configMAX_SYSCALL_INTERRUPT_PRIORITY,
787  and therefore also guaranteed to be invalid.
788 
789  FreeRTOS maintains separate thread and ISR API functions to ensure
790  interrupt entry is as fast and simple as possible.
791 
792  The following links provide detailed information:
793  http://www.freertos.org/RTOS-Cortex-M3-M4.html
794  http://www.freertos.org/FAQHelp.html */
795  configASSERT( ucCurrentPriority >= ucMaxSysCallPriority );
796  }
797

799  that define each interrupt's priority to be split between bits that
800  define the interrupt's pre-emption priority bits and bits that define
801  the interrupt's sub-priority. For simplicity all bits must be defined
802  to be pre-emption priority bits. The following assertion will fail if
803  this is not the case (if some bits represent a sub-priority).
804 
805  If the application only uses CMSIS libraries for interrupt
806  configuration then the correct setting can be achieved on all Cortex-M
807  devices by calling NVIC_SetPriorityGrouping( 0 ); before starting the
808  scheduler. Note however that some vendor specific peripheral libraries
809  assume a non-zero priority group setting, in which cases using a value
810  of zero will result in unpredicable behaviour. */
811  configASSERT( ( portAIRCR_REG & portPRIORITY_GROUP_MASK ) <= ulMaxPRIGROUPValue );
812  }
