92 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
93 {
94  /* Simulate the stack frame as it would be created by a context switch
95  interrupt. */
96  pxTopOfStack--; /* Offset added to account for the way the MCU uses the stack on entry/exit of interrupts. */
97  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
98  pxTopOfStack--;
99  *pxTopOfStack = ( StackType_t ) pxCode; /* PC */
100  pxTopOfStack--;
101  *pxTopOfStack = ( StackType_t ) prvTaskExitError; /* LR */
102  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
103  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
104  pxTopOfStack -= 8; /* R11..R4. */
105 
106  return pxTopOfStack;
107 }

110 static void prvTaskExitError( void )
111 {
112  /* A function that implements a task must not exit or attempt to return to
