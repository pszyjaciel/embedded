533  * This is a common interrupt handling function called for different instances of the APP.

535  * * param[in] handle SPI_SLAVE APP handle pointer of type SPI_SLAVE_t*
536  *
537  * * return void

539 void SPI_SLAVE_lTransmitHandler(const SPI_SLAVE_t * const handle)
540 {
541  SPI_SLAVE_RUNTIME_t * ptr_runtime = handle->runtime;
542  uint32_t bytes_per_word = (uint32_t)ptr_runtime->data_size;
543  uint16_t data_value;
544 
545  if (ptr_runtime->tx_data_index < ptr_runtime->tx_data_count)
546  {
547  if (handle->config->tx_fifo_size != XMC_USIC_CH_FIFO_DISABLED)
548  {
549  /*When Transmit FIFO is enabled*/
550  /*Fill the transmit FIFO */
551  while (XMC_USIC_CH_TXFIFO_IsFull(handle->channel) == false)
552  {
553  if (ptr_runtime->tx_data_index < ptr_runtime->tx_data_count)
554  {
555  memcpy((void *)&data_value, (void *)&ptr_runtime->tx_data[ptr_runtime->tx_data_index * bytes_per_word],
556  bytes_per_word);
557  /*Load the FIFO byte by byte till either FIFO is full or all data is loaded*/
558  XMC_SPI_CH_Transmit(handle->channel,data_value, ptr_runtime->spi_current_mode);
559  ptr_runtime->tx_data_index++;
560  }
561  else
562  {
563  break;
564  }
565  }
566  }
567  else
568  {
569  /*When Transmit FIFO is disabled*/
570  memcpy((void *)&data_value, (void *)&ptr_runtime->tx_data[ptr_runtime->tx_data_index * bytes_per_word], bytes_per_word);
571  XMC_SPI_CH_Transmit(handle->channel,data_value, ptr_runtime->spi_current_mode);
572  ptr_runtime->tx_data_index++;
573  }
574  }
575  else
576  {
577  if (XMC_USIC_CH_TXFIFO_IsEmpty(handle->channel) == true)
578  {
579  if (handle->config->tx_fifo_size != XMC_USIC_CH_FIFO_DISABLED)
580  {

690 /*
691  * Receive interrupt handler for the APP.
692  * This is a common interrupt handling function for different instances of the SPI_SLAVE APP.
693  *
694  * param[in] handle SPI_SLAVE APP handle pointer of type SPI_SLAVE_t*
695  *
696  * return void
697  */
698 void SPI_SLAVE_lReceiveHandler(const SPI_SLAVE_t * const handle)
699 {
700  SPI_SLAVE_RUNTIME_t * ptr_runtime = handle->runtime;
701  uint32_t bytes_per_word = (uint32_t)ptr_runtime->data_size;
702  uint16_t data_value;
703 
704  if (handle->config->rx_fifo_size != XMC_USIC_CH_FIFO_DISABLED)
705  {
706  /*When Receive FIFO is enabled*/
707  while (XMC_USIC_CH_RXFIFO_IsEmpty(handle->channel) == false)
708  {
709  if (ptr_runtime->rx_data_index < ptr_runtime->rx_data_count)
710  {
711  /*Read all the content of Receive FIFO */
712  data_value = XMC_SPI_CH_GetReceivedData(handle->channel);
713  memcpy((void *)&ptr_runtime->rx_data[ptr_runtime->rx_data_index * bytes_per_word], (void *)&data_value, bytes_per_word);
714  ptr_runtime->rx_data_index++;
715  }
716 
717  if (ptr_runtime->rx_data_index == ptr_runtime->rx_data_count)
718  {
719  /*Reception complete*/
720  ptr_runtime->rx_busy = false;
721  /*Disable both standard receive and alternative receive FIFO events*/
722  XMC_USIC_CH_RXFIFO_DisableEvent(handle->channel,
723  (uint32_t)((uint32_t)XMC_USIC_CH_RXFIFO_EVENT_CONF_STANDARD |
724  (uint32_t)XMC_USIC_CH_RXFIFO_EVENT_CONF_ALTERNATE));
725  if (handle->config->rx_cbhandler != NULL)
726  {
727  /*Execute the 'End of reception' callback function*/
728  handle->config->rx_cbhandler();
729  }
730  break;
731  }
732  }
733  /*Set the trigger limit if data still to be received*/
734  if (ptr_runtime->rx_data_index < ptr_runtime->rx_data_count)
735  {
736  SPI_SLAVE_lReconfigureRxFIFO(handle,
737  (uint32_t)(ptr_runtime->rx_data_count - ptr_runtime->rx_data_index));
738  }
739  }
740  else
741  {
742  /*When RxFIFO is disabled*/
743  if (ptr_runtime->rx_data_index < ptr_runtime->rx_data_count)
744  {
745  data_value = XMC_SPI_CH_GetReceivedData(handle->channel);
746  memcpy((void *)&ptr_runtime->rx_data[ptr_runtime->rx_data_index * bytes_per_word], (void *)&data_value, bytes_per_word);
747  ptr_runtime->rx_data_index++;
748  }
749 
750  if (ptr_runtime->rx_data_index == ptr_runtime->rx_data_count)
751  {
752  /*Reception complete*/
753  ptr_runtime->rx_busy = false;
754  /*Disable both standard receive and alternative receive FIFO events*/
755  XMC_SPI_CH_DisableEvent(handle->channel,
756  (uint32_t)((uint32_t)XMC_SPI_CH_EVENT_ALTERNATIVE_RECEIVE | (uint32_t)XMC_SPI_CH_EVENT_STANDARD_RECEIVE));
757 
758  if (handle->config->rx_cbhandler != NULL)
759  {
760  /*Execute the 'End of reception' callback function*/
761  handle->config->rx_cbhandler();
762  }
763  }
764  }

934  memcpy((void *)&rx_dataptr[data_indx - bytes_per_word], (void *)&data_value, bytes_per_word);

936  ptr_runtime->rx_busy = false;
