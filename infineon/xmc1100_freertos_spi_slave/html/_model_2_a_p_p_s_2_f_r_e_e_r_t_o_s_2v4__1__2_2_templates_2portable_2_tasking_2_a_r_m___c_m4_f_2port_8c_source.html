110 StackType_t *pxPortInitialiseStack( StackType_t *pxTopOfStack, TaskFunction_t pxCode, void *pvParameters )
111 {
112  /* Simulate the stack frame as it would be created by a context switch
113  interrupt. */
114 
115  /* Offset added to account for the way the MCU uses the stack on entry/exit
116  of interrupts, and to ensure alignment. */
117  pxTopOfStack--;
118 
119  *pxTopOfStack = portINITIAL_XPSR; /* xPSR */
120  pxTopOfStack--;
121  *pxTopOfStack = ( ( StackType_t ) pxCode ) & portSTART_ADDRESS_MASK; /* PC */
122  pxTopOfStack--;
123  *pxTopOfStack = ( StackType_t ) portTASK_RETURN_ADDRESS; /* LR */
124 
125  /* Save code space by skipping register initialisation. */
126  pxTopOfStack -= 5; /* R12, R3, R2 and R1. */
127  *pxTopOfStack = ( StackType_t ) pvParameters; /* R0 */
128 
129  /* A save method is being used that requires each task to maintain its
130  own exec return value. */
131  pxTopOfStack--;
132  *pxTopOfStack = portINITIAL_EXC_RETURN;
133 
134  pxTopOfStack -= 8; /* R11, R10, R9, R8, R7, R6, R5 and R4. */
135 
136  return pxTopOfStack;
137 }

157 BaseType_t xPortStartScheduler( void )
158 {
159  /* configMAX_SYSCALL_INTERRUPT_PRIORITY must not be set to 0.
160  See http://www.FreeRTOS.org/RTOS-Cortex-M3-M4.html */
161  configASSERT( ( configMAX_SYSCALL_INTERRUPT_PRIORITY ) );
162 
163  /* Make PendSV and SysTick the lowest priority interrupts. */
164  *(portNVIC_SYSPRI2) |= portNVIC_PENDSV_PRI;
165  *(portNVIC_SYSPRI2) |= portNVIC_SYSTICK_PRI;
166 
167  /* Start the timer that generates the tick ISR. Interrupts are disabled
168  here already. */
169  prvSetupTimerInterrupt();
170 
171  /* Initialise the critical nesting count ready for the first task. */
172  ulCriticalNesting = 0;
173 
174  /* Ensure the VFP is enabled - it should be anyway. */
175  vPortEnableVFP();
176 
177  /* Lazy save always. */
178  *( portFPCCR ) |= portASPEN_AND_LSPEN_BITS;
179 
180  /* Start the first task. */
181  vPortStartFirstTask();
182 
183  /* Should not get here! */
184  return 0;
185 }

208 void vPortEnterCritical( void )
209 {
210  portDISABLE_INTERRUPTS();
211  ulCriticalNesting++;
212  __DSB();
213  __ISB();
214 
215  /* This is not the interrupt safe version of the enter critical function so
216  assert() if it is being called from an interrupt context. Only API
217  functions that end in "FromISR" can be used in an interrupt. Only assert if
218  the critical nesting count is 1 to protect against recursive calls if the
219  assert function also uses a critical section. */
220  if( ulCriticalNesting == 1 )
221  {
222  configASSERT( ( ( *(portNVIC_INT_CTRL) ) & portVECTACTIVE_MASK ) == 0 );
223  }
224 }
