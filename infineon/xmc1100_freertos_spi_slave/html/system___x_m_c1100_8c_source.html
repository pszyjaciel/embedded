102 __WEAK void SystemCoreClockSetup(void)
103 {
104  /* Override values of CLOCK_VAL1 and CLOCK_VAL2 defined in vector table */
105  /* MCLK = 32MHz, PCLK = 64MHz */
106 
107  SCU_GENERAL->PASSWD = 0x000000C0UL; /* disable bit protection */
108  SCU_CLK->CLKCR = 0x3FF10100UL;
109  while((SCU_CLK->CLKCR & SCU_CLK_CLKCR_VDDC2LOW_Msk));
110  SCU_GENERAL->PASSWD = 0x000000C3UL; /* enable bit protection */
111 
112  SystemCoreClockUpdate();
113 }

115 __WEAK void SystemCoreClockUpdate(void)
116 {
117  static uint32_t IDIV, FDIV;
118 
119  IDIV = ((SCU_CLK->CLKCR) & SCU_CLK_CLKCR_IDIV_Msk) >> SCU_CLK_CLKCR_IDIV_Pos;
120  FDIV = ((SCU_CLK->CLKCR) & SCU_CLK_CLKCR_FDIV_Msk) >> SCU_CLK_CLKCR_FDIV_Pos;
121 
122  if (IDIV != 0)
123  {
124  /* Fractional divider is enabled and used */
125  SystemCoreClock = ((DCO1_FREQUENCY << 6U) / ((IDIV << 8) + FDIV)) << 1U;
126  }
127  else
128  {
129  /* Fractional divider bypassed. Simply divide DCO_DCLK by 2 */
130  SystemCoreClock = DCO1_FREQUENCY >> 1U;
131  }
132 }
