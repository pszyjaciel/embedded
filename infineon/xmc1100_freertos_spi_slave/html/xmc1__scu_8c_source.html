524 /* API which initializes the clock tree ofthe device */
525 void XMC_SCU_CLOCK_Init(const XMC_SCU_CLOCK_CONFIG_t *const config)
526 {
527  /* Remove protection */
528  XMC_SCU_UnlockProtectedBits();
529 
530 #if (UC_SERIES == XMC14)
531  /* OSCHP source selection - OSC mode */
532 
533  if (config->oschp_mode != XMC_SCU_CLOCK_OSCHP_MODE_DISABLED)
534  {
535  if (OSCHP_GetFrequency() > 20000000U)
536  {
537  SCU_ANALOG->ANAOSCHPCTRL |= (uint16_t)SCU_ANALOG_ANAOSCHPCTRL_HYSCTRL_Msk;
538  }
539 
540  SCU_ANALOG->ANAOSCHPCTRL = (uint16_t)(SCU_ANALOG->ANAOSCHPCTRL & ~(SCU_ANALOG_ANAOSCHPCTRL_SHBY_Msk | SCU_ANALOG_ANAOSCHPCTRL_MODE_Msk)) |
541  config->oschp_mode;
542 
543  do
544  {
545  /* Restart OSC_HP oscillator watchdog */
546  SCU_INTERRUPT->SRCLR1 = SCU_INTERRUPT_SRCLR1_LOECI_Msk;
547 
548  /* Enable OSC_HP oscillator watchdog*/
549  SCU_CLK->OSCCSR &= ~SCU_CLK_OSCCSR_XOWDEN_Msk;
550  SCU_CLK->OSCCSR |= SCU_CLK_OSCCSR_XOWDEN_Msk;
551 
552  /* Wait a few DCO2 cycles for the update of the clock detection result */
553  delay(2500);
554 
555  /* check clock is ok */
556  }
557  while(SCU_INTERRUPT->SRRAW1 & SCU_INTERRUPT_SRRAW1_LOECI_Msk);
558  }
559  else /* (config->oschp_mode == XMC_SCU_CLOCK_OSCHP_MODE_DISABLED) */
560  {
561  SCU_ANALOG->ANAOSCHPCTRL |= SCU_ANALOG_ANAOSCHPCTRL_MODE_Msk;
562  }
563 
564  SCU_ANALOG->ANAOSCLPCTRL = (uint16_t)config->osclp_mode;
565 
566  SCU_CLK->CLKCR1 = (SCU_CLK->CLKCR1 & ~SCU_CLK_CLKCR1_DCLKSEL_Msk) |
567  config->dclk_src;
568 
569 #endif
570  /* Update PCLK selection mux. */
571  SCU_CLK->CLKCR = (SCU_CLK->CLKCR & (uint32_t)~(SCU_CLK_CLKCR_PCLKSEL_Msk | SCU_CLK_CLKCR_RTCCLKSEL_Msk)) |
572  config->rtc_src |
573  config->pclk_src;
574 
575  /* Close the lock opened above. */
576  XMC_SCU_LockProtectedBits();
577 
578  /* Update the dividers now */
579  XMC_SCU_CLOCK_ScaleMCLKFrequency(config->idiv, config->fdiv);
580 
581 }

633 /* This API configures main clock (MCLK) frequency to requested frequency value. */
634 void XMC_SCU_CLOCK_SetMCLKFrequency(uint32_t freq_khz)
635 {
636  uint32_t ratio;
637  uint32_t ratio_int;
638  uint32_t ratio_frac;
639 
640 #if UC_SERIES == XMC14
641  if (((SCU_CLK->CLKCR1) & SCU_CLK_CLKCR1_DCLKSEL_Msk) == 0U)
642  {
643  ratio = DCO1_DIV2_FREQUENCY_KHZ_Q22_10 / freq_khz;
644  }
645  else
646  {
647  ratio = ((OSCHP_GetFrequency() / 1000U) << 10U) / freq_khz;
648  }
649 
650  /* Manage overflow */
651  if (ratio > 0xffffffU)
652  {
653  ratio = 0xffffffU;
654  }
655 
656  ratio_int = ratio >> 10U;
657  ratio_frac = ratio & 0x3ffU;
658 #else
659  ratio = DCO1_DIV2_FREQUENCY_KHZ_Q24_8 / freq_khz;
660 
661  /* Manage overflow */
662  if (ratio > 0xffffU)
663  {
664  ratio = 0xffffU;
665  }
666 
667  ratio_int = ratio >> 8U;
668  ratio_frac = ratio & 0xffU;
669 #endif
670 
671  XMC_SCU_CLOCK_ScaleMCLKFrequency(ratio_int, ratio_frac);
672 }

695  (uint32_t)(fdiv << SCU_CLK_CLKCR_FDIV_Pos) |
696  (uint32_t)(1023UL << SCU_CLK_CLKCR_CNTADJ_Pos);
697 #endif
698 
699  while ((SCU_CLK->CLKCR)& SCU_CLK_CLKCR_VDDC2LOW_Msk)
700  {
701  /* Spin until the core supply stabilizes */
702  }
703 
704  if(curr_idiv <= idiv)
705  {
706  /* Requested IDIV is greater than currently programmed IDIV. So downscale the frequency */
707  XMC_SCU_CLOCK_lFrequencyDownScaling(curr_idiv, idiv);
708  }
709  else
710  {
711  /* Requested IDIV is lower than currently programmed IDIV. So upscale the frequency */
712  XMC_SCU_CLOCK_lFrequencyUpScaling(curr_idiv, idiv);
713  }
714 
715  SCU_CLK->CLKCR = (SCU_CLK->CLKCR & (uint32_t)~(SCU_CLK_CLKCR_IDIV_Msk | SCU_CLK_CLKCR_CNTADJ_Msk)) |
716  (uint32_t)(idiv << SCU_CLK_CLKCR_IDIV_Pos) | (uint32_t)(1023UL << SCU_CLK_CLKCR_CNTADJ_Pos);
717 
718  while ((SCU_CLK->CLKCR) & SCU_CLK_CLKCR_VDDC2LOW_Msk)
719  {
720  /* Wait voltage suply stabilization */
721  }
722 
723  XMC_SCU_LockProtectedBits();
724 
725  SystemCoreClockUpdate();
726

731 {

733  {
734  curr_idiv = (uint32_t)(curr_idiv >> 2UL); /* Divide by 4. */

737  (uint32_t)(curr_idiv << SCU_CLK_CLKCR_IDIV_Pos) |
738  (uint32_t)(1023UL << SCU_CLK_CLKCR_CNTADJ_Pos);
739

741  {
742  /* Wait voltage suply stabilization */
743  }
744  }

748 static void XMC_SCU_CLOCK_lFrequencyDownScaling(uint32_t curr_idiv, uint32_t target_idiv)

759  (uint32_t)(curr_idiv << SCU_CLK_CLKCR_IDIV_Pos) |
760  (uint32_t)(1023UL << SCU_CLK_CLKCR_CNTADJ_Pos);
761

763  {
764  /* Wait voltage suply stabilization */
765  }
766  }
