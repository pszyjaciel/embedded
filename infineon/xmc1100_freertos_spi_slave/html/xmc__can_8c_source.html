103 int32_t XMC_CAN_NODE_NominalBitTimeConfigureEx(XMC_CAN_NODE_t *const can_node,
104  const XMC_CAN_NODE_NOMINAL_BIT_TIME_CONFIG_t *const bit_time_config)
105 {
106  /* Check that the CAN frequency is a multiple of the required baudrate */
107  if ((bit_time_config->can_frequency % bit_time_config->baudrate) == 0)
108  {
109  uint32_t prescaler;
110  uint32_t div8 = 0;
111 
112  /* Calculate the factor between can frequency and required baudrate, this is equal to (prescaler x ntq) */
113  uint32_t fcan_div = bit_time_config->can_frequency / bit_time_config->baudrate;
114 
115  /* start with highest ntq, i.e as much as possible time quanta should be used to construct a bit time */
116  uint32_t ntq = XMC_CAN_NODE_MAX_NTQ;
117  while (ntq >= XMC_CAN_NODE_MIN_NTQ)
118  {
119  /* consider this ntq, only if fcan_div is multiple of ntq */
120  if ((fcan_div % ntq) == 0)
121  {
122  prescaler = fcan_div / ntq;
123  if ((prescaler > 0) && (prescaler <= XMC_CAN_NODE_MAX_PRESCALER))
124  {
125  if (prescaler >= 64)
126  {
127  /* consider prescaler >=64, if it is integer divisible by 8*/
128  if ((prescaler & 0x7U) == 0)
129  {
130  div8 = 1;
131  break;
132  }
133  }
134  else
135  {
136  break;
137  }
138  }
139  }
140  --ntq;
141  }
142 
143  if (ntq >= XMC_CAN_NODE_MIN_NTQ)
144  {
145  uint32_t tseg1 = ((ntq - 1) * bit_time_config->sample_point) / 1000;
146  uint32_t tseg2 = ntq - tseg1 - 1;
147 
148  if ((tseg1 < XMC_CAN_NODE_MIN_TSEG1) || (tseg2 < XMC_CAN_NODE_MIN_TSEG2) || (tseg2 < bit_time_config->sjw))
149  {
150  return XMC_CAN_STATUS_ERROR;
151  }
152 
153  XMC_CAN_NODE_EnableConfigurationChange(can_node);
154 
155  /* Configure bit timing register */
156  can_node->NBTR = (((tseg2 - 1u) << CAN_NODE_NBTR_TSEG2_Pos) & (uint32_t)CAN_NODE_NBTR_TSEG2_Msk) |
157  (((bit_time_config->sjw - 1U) << CAN_NODE_NBTR_SJW_Pos) & (uint32_t)CAN_NODE_NBTR_SJW_Msk) |
158  (((tseg1 - 1U) << CAN_NODE_NBTR_TSEG1_Pos) & (uint32_t)CAN_NODE_NBTR_TSEG1_Msk) |
159  (((prescaler - 1U) << CAN_NODE_NBTR_BRP_Pos) & (uint32_t)CAN_NODE_NBTR_BRP_Msk) |
160  ((div8 << CAN_NODE_NBTR_DIV8_Pos) & (uint32_t)CAN_NODE_NBTR_DIV8_Msk);
161 
162  XMC_CAN_NODE_DisableConfigurationChange(can_node);
163 
164  return XMC_CAN_STATUS_SUCCESS;
165  }
166  }
167 
168  return XMC_CAN_STATUS_ERROR;
169 }

171 /* Baudrate Configuration */
172 void XMC_CAN_NODE_NominalBitTimeConfigure (XMC_CAN_NODE_t *const can_node,
173  const XMC_CAN_NODE_NOMINAL_BIT_TIME_CONFIG_t *const can_bit_time)
174 {
175  uint32_t temp_brp = 12U ;
176  uint32_t temp_tseg1 = 12U;
177  uint32_t best_brp = 0U;
178  uint32_t best_tseg1 = 1U;
179  uint32_t best_tseg2 = 0U;
180  uint32_t best_tbaud = 0U;
181  uint32_t best_error = 10000U;
182 
183  XMC_ASSERT("XMC_CAN_NODE_NOMINAL_BIT_TIME_Configure: rate not supported", (can_bit_time->baudrate < 1000000U) ||
184  (can_bit_time->baudrate >= 100000U));
185  XMC_ASSERT("XMC_CAN_NODE_NOMINAL_BIT_TIME_Configure: fCAN not supported",
186  can_bit_time->can_frequency <= 120000000U);
187  XMC_ASSERT("XMC_CAN_NODE_NOMINAL_BIT_TIME_Configure: fCAN not supported",
188  can_bit_time->can_frequency > 5000000U);
189  XMC_ASSERT("XMC_CAN_NODE_NOMINAL_BIT_TIME_Configure: sample point not supported",
190  (can_bit_time->sample_point < 10000U) && ((can_bit_time->sample_point > 0U)));
191 
192  /*
193  * Bit timing & sampling
194  * Tq = (BRP+1)/Fcan if DIV8 = 0
195  * Tq = 8*(BRP+1)/Fcan if DIV8 = 1
196  * TSync = 1.Tq
197  * TSeg1 = (TSEG1+1)*Tq >= 3Tq
198  * TSeg2 = (TSEG2+1)*Tq >= 2Tq
199  * Bit Time = TSync + TSeg1 + TSeg2 >= 8Tq
200  *
201  * Resynchronization:
202  *
203  * Tsjw = (SJW + 1)*Tq
204  * TSeg1 >= Tsjw + Tprop
205  * TSeg2 >= Tsjw
206  */
207  /* search for best baudrate */
208  for (temp_brp = 1U; temp_brp <= 64U; temp_brp++)
209  {
210 
211  uint32_t f_quanta = (uint32_t)((can_bit_time->can_frequency * 10U) / temp_brp);
212  uint32_t temp_tbaud = (uint32_t)(f_quanta / (can_bit_time->baudrate));
213  uint32_t temp_baudrate;
214  uint32_t error;
215 
216  if((temp_tbaud % 10U) > 5U)
217  {
218  temp_tbaud = (uint32_t)(temp_tbaud / 10U);
219  temp_tbaud++;
220  }
221  else
222  {
223  temp_tbaud = (uint32_t)(temp_tbaud / 10U);
224  }
225 
226  if(temp_tbaud > 0U)
227  {
228  temp_baudrate = (uint32_t) (f_quanta / (temp_tbaud * 10U));
229  }
230  else
231  {
232  temp_baudrate = f_quanta / 10U;
233  temp_tbaud = 1;
234  }
235 
236  if(temp_baudrate >= can_bit_time->baudrate)
237  {
238  error = temp_baudrate - can_bit_time->baudrate;
239  }
240  else
241  {
242  error = can_bit_time->baudrate - temp_baudrate;
243  }
244 
245  if ((temp_tbaud <= 20U) && (best_error > error))
246  {
247  best_brp = temp_brp;
248  best_tbaud = temp_tbaud;
249  best_error = (error);
250 
251  if (error < 1000U)
252  {
253  break;
254  }
255  }
256  }
257  /* search for best sample point */
258  best_error = 10000U;
259 
260  for (temp_tseg1 = 64U; temp_tseg1 >= 3U; temp_tseg1--)
261  {
262  uint32_t tempSamplePoint = ((temp_tseg1 + 1U) * 10000U) / best_tbaud;
263  uint32_t error;
264  if (tempSamplePoint >= can_bit_time->sample_point)
265  {
266  error = tempSamplePoint - can_bit_time->sample_point;
267  }
268  else
269  {
270  error = can_bit_time->sample_point - tempSamplePoint;
271  }
272  if (best_error > error)
273  {
274  best_tseg1 = temp_tseg1;
275  best_error = error;
276  }
277  if (tempSamplePoint < (can_bit_time->sample_point))
278  {
279  break;
280  }
281  }
282 
283  best_tseg2 = best_tbaud - best_tseg1 - 1U;
284 
285  XMC_CAN_NODE_EnableConfigurationChange(can_node);
286  /* Configure bit timing register */
287  can_node->NBTR = (((uint32_t)(best_tseg2 - 1u) << CAN_NODE_NBTR_TSEG2_Pos) & (uint32_t)CAN_NODE_NBTR_TSEG2_Msk) |
288  ((((uint32_t)((uint32_t)(can_bit_time->sjw)-1U) << CAN_NODE_NBTR_SJW_Pos)) & (uint32_t)CAN_NODE_NBTR_SJW_Msk)|
289  (((uint32_t)(best_tseg1-1U) << CAN_NODE_NBTR_TSEG1_Pos) & (uint32_t)CAN_NODE_NBTR_TSEG1_Msk)|
290  (((uint32_t)(best_brp - 1U) << CAN_NODE_NBTR_BRP_Pos) & (uint32_t)CAN_NODE_NBTR_BRP_Msk)|
291  (((uint32_t)0U << CAN_NODE_NBTR_DIV8_Pos) & (uint32_t)CAN_NODE_NBTR_DIV8_Msk);
292  XMC_CAN_NODE_DisableConfigurationChange(can_node);
293 }
