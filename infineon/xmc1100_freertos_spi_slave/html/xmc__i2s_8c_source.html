89 {
90  XMC_ASSERT("XMC_I2S_CH_Init: data_delayed_sclk_periods value not valid",(config->data_delayed_sclk_periods > 0U) &&
91  (config->data_delayed_sclk_periods < config->frame_length));
92  XMC_USIC_CH_Enable(channel);
93 
94  if(config->bus_mode == XMC_I2S_CH_BUS_MODE_MASTER)
95  {
96  /* Configure baud rate */
97  (void)XMC_I2S_CH_SetBaudrate(channel, config->baudrate);
98  }
99  /* Configuration of USIC Shift Control */
100  /* Transmission Mode (TRM) = 1 */
101  channel->SCTR = (uint32_t)(0x3UL << USIC_CH_SCTR_TRM_Pos) |
102  (uint32_t)((config->frame_length -1U) << USIC_CH_SCTR_FLE_Pos) |
103  (uint32_t)((config->data_bits - 1U) << USIC_CH_SCTR_WLE_Pos) |
104  USIC_CH_SCTR_SDIR_Msk;
105 
106  /* Configuration of USIC Transmit Control/Status Register */
107  /* TBUF Data Enable (TDEN) = 1 */
108  /* TBUF Data Single Shot Mode (TDSSM) = 1 */
109  /* WA mode enabled(WAMD) = 1 */
110  channel->TCSR = (uint32_t)((channel->TCSR & (~(USIC_CH_TCSR_WLEMD_Msk |
111  USIC_CH_TCSR_SELMD_Msk |
112  USIC_CH_TCSR_FLEMD_Msk |
113  USIC_CH_TCSR_HPCMD_Msk))) |
114  USIC_CH_TCSR_WAMD_Msk |
115  (0x01UL << USIC_CH_TCSR_TDEN_Pos) |
116  USIC_CH_TCSR_TDSSM_Msk);
117 
118  if(config->bus_mode == XMC_I2S_CH_BUS_MODE_MASTER)
119  {
120  /* Configuration of Protocol Control Register */
121  channel->PCR_IISMode = (uint32_t)USIC_CH_PCR_IISMode_WAGEN_Msk;
122  }
123 
124  /* Configuration of Protocol Control Register */
125  channel->PCR_IISMode |= (uint32_t)(USIC_CH_PCR_IISMode_DTEN_Msk |
126  (uint32_t)config->wa_inversion) |
127  ((uint32_t)((uint32_t)config->data_delayed_sclk_periods - 1U) << USIC_CH_PCR_IISMode_TDEL_Pos);
128 
129  XMC_I2S_CH_SetSystemWordLength(channel, config->frame_length);
130 
131  /* Clear protocol status */
132  channel->PSCR = 0xFFFFFFFFUL;
133 }

162  uint8_t dctq = 1U;
163  uint8_t pctq = 1U;
164  uint8_t best_error = 64U;
165  uint8_t error;
166  XMC_ASSERT("XMC_I2S_CH_Init: data_delayed_sclk_periods value not valid",(sclk_cycles_system_word_length > 0U) && (sclk_cycles_system_word_length < 65U));
167 
168 
169  for (dctq_temp =1U; dctq_temp < 33U ; dctq_temp++)
170  {
171  for (pctq_temp =1U; pctq_temp < 5U ; pctq_temp++)
172  {
173  sclk_cycles_system_word_length_temp = ((uint32_t)dctq_temp) * ((uint32_t)pctq_temp);
174  if(sclk_cycles_system_word_length_temp == sclk_cycles_system_word_length)
175  {
176  dctq = dctq_temp;
177  pctq = pctq_temp;
178  break;
179  }
180  if (sclk_cycles_system_word_length_temp > sclk_cycles_system_word_length)
181  {
182  error = (uint8_t)(sclk_cycles_system_word_length_temp - sclk_cycles_system_word_length);
183  }
184  else
185  {
186  error = (uint8_t)(sclk_cycles_system_word_length - sclk_cycles_system_word_length_temp);
187  }
188 
189  if(error < best_error)
190  {
191  best_error = error;
192  dctq = dctq_temp;
193  pctq = pctq_temp;
194  }
195  }
196  }
197  channel->BRG = (uint32_t)((channel->BRG & ~((uint32_t)(USIC_CH_BRG_DCTQ_Msk |
198  USIC_CH_BRG_PCTQ_Msk))) |
199  (uint32_t)((uint32_t)((uint32_t)((uint32_t)dctq- 1U) << USIC_CH_BRG_DCTQ_Pos) |

203 /* Puts the data into FIFO if FIFO mode is enabled or else into standard buffers, by setting the proper mode. */
204 void XMC_I2S_CH_Transmit(XMC_USIC_CH_t *const channel, const uint16_t data, const XMC_I2S_CH_CHANNEL_t channel_number)
205 {
206  /* Check FIFO size */
207  if ((channel->TBCTR & USIC_CH_TBCTR_SIZE_Msk) == 0U)
208  {
209  while((uint32_t)XMC_USIC_CH_GetTransmitBufferStatus(channel) == (uint32_t)XMC_USIC_CH_TBUF_STATUS_BUSY)
210  {
211  }
212 
213  XMC_I2S_CH_ClearStatusFlag(channel, (uint32_t)XMC_I2S_CH_STATUS_FLAG_TRANSMIT_BUFFER_INDICATION);
214 
215  channel->TBUF[(uint32_t)channel_number << 4] = data;
216  }
217  else
218  {
219  channel->IN[(uint32_t)channel_number << 4] = data;
220  }
221 }
