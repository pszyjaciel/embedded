78 void XMC_UART_CH_Init(XMC_USIC_CH_t *channel, const XMC_UART_CH_CONFIG_t *const config)
79 {
80  uint32_t oversampling = XMC_UART_CH_OVERSAMPLING;
81 
82  /* USIC channel switched on*/
83  XMC_USIC_CH_Enable(channel);
84 
85  if(config->oversampling != 0U)
86  {
87  oversampling = (uint32_t)config->oversampling;
88  }
89 
90  /* Configure baud rate */
91  (void)XMC_USIC_CH_SetBaudrate(channel, config->baudrate, oversampling);
92 
93  /* Configure frame format
94  * Configure the number of stop bits
95  * Pulse length is set to 0 to have standard UART signaling,
96  * i.e. the 0 level is signaled during the complete bit time
97  * Sampling point set equal to the half of the oversampling period
98  * Enable Sample Majority Decision
99  * Enable Transfer Status BUSY
100  */
101  channel->PCR_ASCMode = (uint32_t)(((config->stop_bits - 1UL) << USIC_CH_PCR_ASCMode_STPB_Pos) |
102  (((oversampling >> 1UL) + 1UL) << USIC_CH_PCR_ASCMode_SP_Pos) |
103  USIC_CH_PCR_ASCMode_SMD_Msk |
104  USIC_CH_PCR_ASCMode_RSTEN_Msk | USIC_CH_PCR_ASCMode_TSTEN_Msk);
105 
106  /* Set passive data level, high
107  Set word length. Data bits - 1
108  If frame length is > 0, frame_lemgth-1; else, FLE = WLE (Data bits - 1)
109  Transmission Mode: The shift control signal is considered active if it
110  is at 1-level. This is the setting to be programmed to allow data transfers */
111  channel->SCTR = (uint32_t)((((uint32_t)config->data_bits - 1UL) << USIC_CH_SCTR_WLE_Pos) |
112  ((0x1UL << USIC_CH_SCTR_TRM_Pos) | USIC_CH_SCTR_PDL_Msk));
113 
114  if (config->frame_length != 0U)
115  {
116  channel->SCTR |= (uint32_t)(((uint32_t)config->frame_length - 1UL) << USIC_CH_SCTR_FLE_Pos);
117  }
118  else
119  {
120  channel->SCTR |= (uint32_t)(((uint32_t)config->data_bits - 1UL) << USIC_CH_SCTR_FLE_Pos);
121  }
122 
123  /* Enable transfer buffer */
124  channel->TCSR = (0x1UL << USIC_CH_TCSR_TDEN_Pos) |
125  USIC_CH_TCSR_TDSSM_Msk;
126 
127  /* Clear protocol status */
128  channel->PSCR = 0xFFFFFFFFUL;
129 
130  /* Set parity settings */
131  channel->CCR = (uint32_t)config->parity_mode;
132 }
