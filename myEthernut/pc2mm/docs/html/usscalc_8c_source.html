115 void prepTelegram(uint16_t myRegNo)
116 {
117  // indicates the start of a message
118  myTelegram[0] = 0x02;
119  myTelegram[1] = 0x0c;
120 
121  // address of the slave node
122  myTelegram[2] = (0x00 | (mirror << 6)); // the number of RS485 slave-device
123 
124  myRegNo = myRegNo | 0x1000; // set 'response' bit
125  myTelegram[3] = ((0 | myRegNo | (AK << 12) | (SP << 11)) & 0xff00) >> 8;
126  myTelegram[4] = (0 | myRegNo | (AK << 12) | (SP << 11)) & 0x00ff; // AK
127 
128  // PKW, 2. word - be carefull with bit15 in PNUex
129  // bit15-8: page index, bit7-0: subindex
130  myTelegram[5] = ((PNUex << 12) & 0xff00) >> 8;
131  myTelegram[6] = IDX & 0x00ff; // command
132 
133  // PKW, 3. word = PWE1 = 1st parameter value, high-order word
134  myTelegram[7] = 0x00;
135  myTelegram[8] = 0x00;
136 
137  // PKW, 4. word = PWE2 = 2nd parameter value, low-order word
138  myTelegram[9] = 0x00;
139  myTelegram[10] = 0x00;
140  // ----------------- end of PKW -----------------
141 
142  // ----------------- start of PZD -----------------
143  // PZD
144  // The number of PZD words in a USS-telegram is determined by parameter P2012.
145  // Request to USS slave: Control word 1(STW1) and main setpoint (HSW)
146  // Response to USS master: Status word 1 (ZSW1) and actual value (HIW)
147  myTelegram[11] = 0x00; // przy odpowiedzi to ZSW
148  myTelegram[12] = 0x00;
149 
150  // control sum
151  myTelegram[13] = calcBCC();
152 }
